CREATE TABLE `tree` (
 `ID` varchar(50) NOT NULL COMMENT 'Taxon id',
 `ScientificNameID` varchar(50) NOT NULL COMMENT 'the id of the scientific name regestered in "Names of Life"',
 `CanonicalName` varchar(100) NOT NULL COMMENT 'the typicla type of scientific name without author infromation',
 `ParentID` varchar(50) NOT NULL COMMENT 'Indicate the taxon'' parent',
 `AuthorInfo` varchar(100) DEFAULT NULL COMMENT 'Author information of the scientific name',
 `TreeID` varchar(50) NOT NULL COMMENT 'Indicate the tree which the taxon belongs to',
 `AccordingTo` varchar(100) DEFAULT NULL COMMENT '=sec. means this taxon was proposed in or by',
 `Rank` varchar(50) DEFAULT NULL COMMENT 'Rank the the taxon like family, genus,species...',
 `sourceID` varchar(50) DEFAULT NULL COMMENT 'id in the original source',
 `Ancestors` varchar(500) DEFAULT NULL COMMENT 'A string composed by the parents'' names of this taxon.',
 PRIMARY KEY (`ID`),
 UNIQUE KEY `ID` (`ID`),
 KEY `TreeID` (`TreeID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='Each record of ''tree'' means a taxon including its propertis such as name, rank, position and so on.';
